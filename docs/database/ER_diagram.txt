Table character {
  id integer [pk]
  name string
  status string
  species string
  type string
  gender string
  origin_id integer
  last_location_id integer
  image string
  created_at timestamp
  updated_at timestamp
}

Table location {
  id integer [pk]
  name string
  type string
  dimension string
  created_at timestamp
  updated_at timestamp
}

Table episode {
  id integer [pk]
  name string
  air_date timestamp
  episode string
  created_at timestamp
  updated_at timestamp
}

Table episode_character {
  episode_id integer
  character_id integer
}

Ref: episodes.id < episode_characters.episode_id

Ref: characters.id < episode_characters.character_id

Ref: characters.origin_id > locations.id

Ref: characters.last_location_id > locations.id
