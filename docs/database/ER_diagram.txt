Table character {
  id integer [pk]
  name string
  status string
  species string
  type string
  gender string
  origin_id integer
  last_location_id integer
  image string
  created_at timestamp
  updated_at timestamp
}

Table location {
  id integer [pk]
  name string
  type string
  dimension string
  created_at timestamp
  updated_at timestamp
}

Table episode {
  id integer [pk]
  name string
  air_date timestamp
  code string
  created_at timestamp
  updated_at timestamp
}

Table episode_character {
  episode_id integer
  character_id integer
}

Ref: episode.id < episode_character.episode_id

Ref: character.id < episode_character.character_id

Ref: character.origin_id > location.id

Ref: character.last_location_id > location.id
